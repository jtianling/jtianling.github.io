typedef vector<sipair> sidv;

{  
    string stra;  
    int ia;  
    sipair paira,pairb;  
    sidv sidva;  
    while(cin>>stra>>ia)  
    {  
        paira.first = stra;  
        paira.second = ia;    //第一种方法  
        pairb = make_pair(stra,ia);  //第二种方法  
        sipair pairc(paira);

        sidva.push_back(paira);  
    }  
    for(sidv::iterator it = sidva.begin();it != sidva.end();++it)  
    {  
        cout<<"the pair is:"  
            <<it->first<<" "<<it->second<<endl;  
    }  
      
    return 0;  
}
